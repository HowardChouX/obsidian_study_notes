# switch

```c++
[[include]] <bits/stdc++.h>

using namespace std;

  

int main()

{

    char grade;

    cin >> grade;

    switch (grade)

    {

    case 'A':

        cout << "85~100分\n";

        break;

    case 'B':

        cout << "70~84分\n";

        break;

    case 'C':

        cout << "60~69分\n";

        break;

    case 'D':

        cout << "50~59分\n";

        break;

    default:

        cout << "低于50分\n";

        break;

    }

    return 0;

}
```
